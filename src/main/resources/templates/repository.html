New Repository
                
                    
git clone git@gitlab.com:/.git

                    
cd test

                    
touch README.md

                    
git add README.md

                    
git commit -m "add README"

                    
git push -u origin master
Existing Folder
                
                    
cd existing_folder

                    
git init

                    
git remote add origin git@gitlab.com:/.git

                    
git add .

                    
git commit -m "Initial commit"

                    
git push -u origin master
